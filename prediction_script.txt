df.dropna(subset='effective_yr', inplace = True)

df[df['prod_year'].isna()]

stat_cols = ['effective_yr', 'insured_value', 'premium', 'seats_num','carrying_capacity','ccm_ton' ,'claim_paid', 'weighted_exposure']

df[stat_cols].describe().T




df2 = df[['sex', 'insr_type','effective_yr', 'premium', 'seats_num','carrying_capacity','prod_year', 'type_vehicle', 'ccm_ton', 'make', 'usage','claim_paid','weighted_exposure']]


df2.drop_duplicates(inplace = True)



df2 = df2[df2['weighted_exposure'] > 0]


x = df2[['sex', 'insr_type','effective_yr', 'premium', 'seats_num','carrying_capacity','prod_year', 'type_vehicle', 'ccm_ton', 'make', 'usage','claim_paid']]
y = df2['weighted_exposure']


from sklearn.pipeline import Pipeline 
from sklearn.preprocessing import StandardScaler, OrdinalEncoder,FunctionTransformer
from sklearn.model_selection import train_test_split
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer



num_cols = x.select_dtypes(include=['number']).columns

cat_cols = x.select_dtypes(include=['object']).columns


num_pipe = Pipeline([
    ('imputer', SimpleImputer(strategy = 'median')),
    ('scaler', StandardScaler())
])


cat_pipe = Pipeline([
    ('imputer', SimpleImputer(strategy = 'most_frequent')),
    ('to_string', FunctionTransformer(lambda x: x.astype(str))),
    ('encoder', OrdinalEncoder(handle_unknown = 'use_encoded_value', unknown_value = -50))
])

x_train, x_test, y_train, y_test = train_test_split(x, y, test_size = 0.25, random_state = 45)

processor = ColumnTransformer(
    transformers=[
        ('num_pipe', num_pipe, num_cols),
        ('cat_pipe', cat_pipe, cat_cols)
    ],
    remainder = 'drop'
)

processor 



from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor
from sklearn.svm import SVR
from sklearn.tree import DecisionTreeRegressor



lr_pipe = Pipeline([
    ('processor', processor),
    ('lr_model', LinearRegression())
])

lr_pipe.fit(x_train, y_train)

lr_predict = lr_pipe.predict(x_test)


rf_pipe = Pipeline([
    ('processor', processor),
    ('rf_model', RandomForestRegressor())
])

rf_pipe.fit(x_train, y_train)
rf_predict = rf_pipe.predict(x_test)



from sklearn.metrics import mean_squared_error, mean_absolute_error,  mean_absolute_percentage_error,r2_score
print(mean_squared_error(y_test, lr_predict))
print(mean_absolute_error(y_test, lr_predict))
print(mean_absolute_percentage_error(y_test, lr_predict))
print(r2_score(y_test, lr_predict))



print(mean_squared_error(y_test, rf_predict))
print(mean_absolute_error(y_test, rf_predict))
print(mean_absolute_percentage_error(y_test, rf_predict))
print(r2_score(y_test, rf_predict))




rf_importance = rf_pipe.named_steps['rf_model'].feature_importances_
rf_importance = pd.DataFrame({'feature': x_train.columns, 'importance': rf_importance})
rf_importance = rf_importance.sort_values(by = 'importance', ascending = False)

print(rf_importance)



import matplotlib.pyplot as plt

sns.barplot(x = 'importance', y = 'feature', data = rf_importance.sort_values(by = 'importance', ascending = False))
